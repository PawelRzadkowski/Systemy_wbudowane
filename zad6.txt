int main(void) {

    unsigned long adc;
    unsigned long low = 1023 / 3;
    unsigned long high = low * 2;

    system_init();

    adc = ADC_Read10bit(ADC_CHANNEL_POTENTIOMETER);

    if (adc < low) {

        time1 = 60;
        time2 = 60;
    }

    else if (adc < high) {

        time1 = 60 * 3;
        time2 = 60 * 3;
    }

    else {

        time1 = 60 * 5;
        time2 = 60 * 5;
    }

    start = 1;

    game_loop();

    return 0;
}
